5def search_v1(token, query_str): 

6 """ 

7 Given a query string, return a collection of messages in all of the channels/ 

8 DMs that the user has joined that contain the query (case-insensitive). 

9 There is no expected order for these messages. 

10 

11 Arguments: 

12 token (str) - A registered user's token 

13 query_str (str) - The pattern that the user wants to find in the message collection 

14 

15 Exceptions: 

16 InputError - When length of query_str is less than 1 OR over 1000 characters 

17 AccessError - Thrown when the token passed in is invalid. 

18 

19 Return Value: 

20 Returns { messages } given a valid token str is inputted. 

21 messages will be a list of dictionaries >> [{},{},{}], 

22 where each dictionary contains types { message_id, u_id, message, time_sent, reacts, is_pinned } 

23  

24 """ 

25 db = data_store.get() 

26 user = extract_validate_token_user_id(db, token) 

27 

28 # Check that the query string is an acceptable length 

29 if len(query_str) < 1: 

30 raise InputError("query_str needs to be at least 1 character long") 

31 if len(query_str) > 1000: 

32 raise InputError("query_str can only be at most be 1000 characters long") 

33 

34 msg_ids_to_be_examined = [] 

35 channels = db['channels'] 

36 for _, channel in channels.items(): 

37 if user in channel['members']: 

38 msg_ids_to_be_examined += channel['messages'] 

39 dms = db['dms'] 

40 for _, dm in dms.items(): 

41 if user in dm['members']: 

42 msg_ids_to_be_examined += dm['messages'] 

43 

44 

45 pattern_occurences = [] 

46 messages = db['messages'] 

47 for m_id, msg in messages.items(): 

48 if m_id in msg_ids_to_be_examined and query_str.lower() in msg['message'].lower(): 

49 pattern_occurences.append(msg) 

50 

51 return {'messages': pattern_occurences}
